#先安装zsh   然后安装pyenv更方便

zsh

1.安装zsh
sudo apt-get update
sudo apt-get install zsh

2.安装Oh My Zsh(可直接通过复制脚本内容来安装)
sh -c "$(curl -fsSL https://raw.github.com/ohmyzsh/ohmyzsh/master/tools/install.sh)"

3.安装zsh-autosuggestions插件和zsh-syntax-highlighting插件：
git clone https://github.com/zsh-users/zsh-autosuggestions ${ZSH_CUSTOM:-~/.oh-my-zsh/custom}/plugins/zsh-autosuggestions
git clone https://github.com/zsh-users/zsh-syntax-highlighting.git ${ZSH_CUSTOM:-~/.oh-my-zsh/custom}/plugins/zsh-syntax-highlighting

4.打开~/.zshrc文件，并在plugins=(...)部分添加zsh-syntax-highlighting和 zsh-autosuggestions
plugins=(git zsh-autosuggestions zsh-syntax-highlighting z)

5.重加载zshrc
source ~/.zshrc



pyenv


1.安装依赖项
sudo apt-get update
sudo apt-get install -y build-essential libssl-dev zlib1g-dev libbz2-dev \
    libreadline-dev libsqlite3-dev wget curl llvm libncurses5-dev libncursesw5-dev \
    xz-utils tk-dev libffi-dev liblzma-dev python3-openssl git

2.安装pyenv
2.1.curl https://pyenv.run | bash
2.1 git clone https://github.com/pyenv/pyenv.git ~/.pyenv

3.添加path
for bash

echo 'export PYENV_ROOT="$HOME/.pyenv"' >> ~/.bashrc
echo 'export PATH="$PYENV_ROOT/bin:$PATH"' >> ~/.bashrc
echo -e 'if command -v pyenv 1>/dev/null 2>&1; then\n  eval "$(pyenv init --path)"\nfi' >> ~/.bashrc

for zash

echo 'export PYENV_ROOT="$HOME/.pyenv"' >> ~/.zshrc
echo 'export PATH="$PYENV_ROOT/bin:$PATH"' >> ~/.zshrc
echo -e 'if command -v pyenv 1>/dev/null 2>&1; then\n  eval "$(pyenv init --path)"\nfi' >> ~/.zshrc

4.重启shell
exec $SHELL

5.安装pyenv-virtualenv
git clone https://github.com/pyenv/pyenv-virtualenv.git $(pyenv root)/plugins/pyenv-virtualenv

6.初始化   
echo 'eval "$(pyenv virtualenv-init -)"' >> ~/.bashrc        #for bash
echo 'eval "$(pyenv virtualenv-init -)"' >> ~/.zshrc         # for zash 
7.重启shell
exec $SHELL
8.修改pip 安装源
mkdir ~/.pip
nano ~/.pip/pip.conf

添加
[global]
index-url = https://pypi.tuna.tsinghua.edu.cn/simple
9.验证
pyenv --version

10.安装GCC（GNU编译器集合）
sudo apt-get update
sudo apt-get install build-essential

11.修改python安装源，在bashrc或者zshrc中添加以下内容后source ...
export PYTHON_BUILD_MIRROR_URL_SKIP_CHECKSUM=1
export PYTHON_BUILD_MIRROR_URL="https://registry.npmmirror.com/-/binary/python"

12  安装python
pyenv install ...
